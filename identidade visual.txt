        // Micro-interactions

            btn.addEventListener('click', function(e) {
                if (!this.classList.contains('cursor-not-allowed')) {
                    const ripple = document.createElement('div');
                    ripple.className = 'absolute inset-0 bg-white/20 animate-ping rounded-full pointer-events-none';
                    this.appendChild(ripple);
                    setTimeout(() => ripple.remove(), 600);
                }
            });
        });

        // Simple active state toggle for calendar
        const calendarButtons = document.querySelectorAll('.grid button:not(.cursor-not-allowed)');
        calendarButtons.forEach(btn => {
            btn.addEventListener('click', () => {
                calendarButtons.forEach(b => {
                    b.classList.remove('bg-primary-container', 'text-on-primary-container', 'font-bold', 'shadow-[0_4px_12px_rgba(204,255,0,0.4)]');
                    b.classList.add('hover:bg-surface-container-high');
                });
                btn.classList.add('bg-primary-container', 'text-on-primary-container', 'font-bold', 'shadow-[0_4px_12px_rgba(204,255,0,0.4)]');
